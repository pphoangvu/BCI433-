                Select
                   When       (&Option = '1' *or &option = '01') CallSubr OPTION1
                   When       (&Option = '2' *or &option = '02') CallSubr OPTION2
                   When       (&Option = '3' *or &option = '03') CallSubr OPTION3
                   When       (&Option = '4' *or &option = '04') CallSubr OPTION4
